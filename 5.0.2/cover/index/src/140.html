using Hazelcast.Serialization;

namespace Hazelcast.Query
{
    internal class CaseInsensitiveLikePredicate : LikePredicate, IIdentifiedDataSerializable
    {
        public CaseInsensitiveLikePredicate()
        { }

        public CaseInsensitiveLikePredicate(string attributeName, string expression)
            : base(attributeName, expression)
        { }

        public override int ClassId => PredicateDataSerializerHook.ILikePredicate;

        public override string ToString()
        {
            return "ILIKE(" + AttributeName + ", '" + Expression + "')";
        }
    }
}
