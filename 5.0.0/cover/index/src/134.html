using System;

namespace Hazelcast.Query
{
    internal class AndPredicate : LogicalPredicateBase
    {
        public AndPredicate()
            : this(Array.Empty<IPredicate>())
        { }

        public AndPredicate(params IPredicate[] predicates)
            : base(predicates)
        { }

        public AndPredicate Concat(IPredicate predicate)
            => new AndPredicate(ConcatInternal(predicate));

        public override int ClassId => PredicateDataSerializerHook.AndPredicate;
    }
}
