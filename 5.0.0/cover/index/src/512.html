namespace Hazelcast.Metrics
{
    // an abstract metric (without the value)
    internal abstract class Metric
    {
        public IMetricDescriptor Descriptor { get; set; }

        public static Metric<T> Create<T>(IMetricDescriptor descriptor, T value)
            => new Metric<T> { Descriptor = descriptor, Value = value };

        public abstract string StringValue { get; }
    }
}
