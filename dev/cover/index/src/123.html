namespace Hazelcast.Query
{
    internal class NotEqualPredicate : EqualPredicate
    {
        public NotEqualPredicate()
        { }

        public NotEqualPredicate(string attributeName, object value)
            : base(attributeName, value)
        { }

        public override int ClassId => PredicateDataSerializerHook.NotEqualPredicate;

        public override string ToString()
        {
            return "(" + AttributeName + " != " + Value + ")";
        }
    }
}
