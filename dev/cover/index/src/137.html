using System.Collections.Generic;
using Hazelcast.Exceptions;

namespace Hazelcast.Protocol.Models
{
    internal class ErrorHolder
    {
        public int ErrorCode { get; }
        public string ClassName { get; }
        public string Message { get; }
        public IEnumerable<StackTraceElement> StackTraceElements { get; }

        public ErrorHolder(RemoteError errorCode, string className, string message, IEnumerable<StackTraceElement> stackTraceElements)
        {
            ErrorCode = (int) errorCode;
            ClassName = className;
            Message = message;
            StackTraceElements = stackTraceElements;
        }

        public ErrorHolder(int errorCode, string className, string message, IEnumerable<StackTraceElement> stackTraceElements)
        {
            ErrorCode = errorCode;
            ClassName = className;
            Message = message;
            StackTraceElements = stackTraceElements;
        }
    }
}
