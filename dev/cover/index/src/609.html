using System.Threading;

namespace Hazelcast.Core
{
    /// <summary>
    /// Implements an <see cref="ISequence{Int64}" />.
    /// </summary>
    internal class Int64Sequence : ISequence<long>
    {
        private long _value;

        /// <inheritdoc />
        public long GetNext()
        {
            return Interlocked.Increment(ref _value);
        }
    }
}
